FROM            dbo.userCategoriesP INNER JOIN
                         dbo.CategoriesP ON dbo.userCategoriesP.CategoryID = dbo.CategoriesP.CategoryId INNER JOIN
                         dbo.UsersP ON dbo.userCategoriesP.UserID = dbo.UsersP.userId
